== create a user account

[source,shell]
----
$NAME=fatima
useradd $NAME
----

Set a password. 

[source,shell]
----
passwd --stdin $NAME
----


Give the account administrative access. 

[source,shell]
----
usermod -a -G wheel $NAME
----

Create a few accounts. 
Gvie them all the same password. 

[source,shell]
----
cat names.txt 
user1
user2 
user3

for NAME in $(cat names.txt)
do
  useradd $NAME
  echo 'Password;1' | passwd --stdin $NAME
done 
----
